def get_more
  current_options = params.inject({}){ |memo, (k,v)| memo[k.to_sym] = v; memo }
  if params.has_key? :persona_range then
    @persona = Persona.find(params[:persona_range]) 
  end
  @tags = Photo.get_tags(current_options)

  respond_to do |format|
    format.html
    format.js
  end
end
